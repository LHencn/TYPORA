6、普通的成员方法根据类走，虚函数成员方法根据对象走

```c++
class A {
public :
    virtual void test() {
        cout << "A" << endl;
    }
    void test() {
        cout << "A" << endl;
    }
};
class B : public A {
public :
    virtual void test() {
        cout << "B" << endl;
    }
    void test() {
        cout << "B" << endl;
    }
};

int main() {
    B b;
    A *p = &b;
    b.test();
    b.test1();
    p->test();
    p->test1();
    return 0;
}
```
